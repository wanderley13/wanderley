PRIMEIRO CODIGO:esse codigo calcula o juros do cartao de uma loja

#implementação de juros
if quantidade_parcela < 4:
juros = (0/100)
elif quantidade_parcela >= 4 < 6:
juros = (4/100)
elif quantidade_parcela >= 6 < 9:
juros = (8 / 100)
elif quantidade_parcela >= 9 < 13:
juros = (16 / 100)
else:
juros = (32/100)
#implementação do valor da parcela e valor total de parcelas
valor_da_parcela = valor_do_pedido * (1 + juros) / quantidade_parcela
valor_total_parcelas = valor_da_parcela * quantidade_parcela
print (valor_da_parcela)
print (valor_total_parcelas)
